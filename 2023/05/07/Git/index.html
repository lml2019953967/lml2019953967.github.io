Git

#配置用户信息
git config -l 
git config --global user.name dragon
git config --global user.email 2019953967@qq.com 
 
#help
git --help [name]

#为常用指令配置别名
touch ~/.bashrc

#初始化git仓库
git init

#工作区-->暂存区
git add


#暂存区-->本地仓库
git commit [-m] [添加注释] 

#添加所有到暂存区
git add .   

#查看仓库状态
git status

#查看提交记录
git log
git log --pretty=oneline --abbrev-commit --all --graph [命令太长，可配置别名]

#版本回退(硬回退)
git reset --hard commitid

#添加文件至忽略列表
touch .gitignore

#查看分支
git branch 

#创建分支
git branch [name]

#切换分支
git chechout [name]

#合并分支
git merge [name]

#删除分支(-D,强制删除)
git branch -d [name]
git branch -D [name]

#查看blob,tree,commit,tag对象
git cat-file -t [name]  ---类型
git cat-file -p [name]  ---内容

#查看最近comit
git reflog

#对比工作区和索引区的不同
git diff

#对比仓库和索引区的不同
git diff --cached

#查看远程仓库
git remote
git remote -v
git remote show origin

#合并远程仓库
git fetch
git merge [commit]

#拉远程仓库到本地
git pull

#推本地仓库到远程仓库并关联本地
git -u push origin [commit]

#git删除远程仓库里分支
git push origin -d [commit]

#查看都有哪些分支
git branch -vv
git branch -a

#工作时暂停,切换分支
git stash
git stash pop(恢复)

git worktree add ../filefolder [branch]
git worktree list
git worktree remove filefolder

#合并Fix bug ,不引入(操作称作backport)
git cherry-pick [commit id]
